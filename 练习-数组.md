##### 选择排序 降序 （选择排序）
```js
let arr=[45,78,9,84,45,75,6]
for(let i=0;i<arr.length-1;i++){
    let maxIndex =i;
    for(let j=i+1;j<arr.length-1;j++){
        if(arr[maxIndex]<arr[j]){
            maxIndex=j
        }
    }
    if(arr[i]<arr[maxIndex]){      //i!=maxIndex
        let temp=arr[maxIndex]
        arr[maxIndex]=arr[i]
        arr[i]=temp
    }
}
console.log(arr)
```

##### 实现随机打乱数组元素的功能，将数组乱序排序。
```js
//核心思想：让元素的下标不固定，也就是元素在数组中的位置不固定，用到了随机数Math.random()  0-1
let arr = [1,2,3];
for (let i = 0; i < arr.length; i++) {
    let index = parseInt(Math.random() * arr.length);   //获取一个随机的下标
    let temp = arr[index];  //将随机下标所对应的元素保存下来
    arr[index] = arr[i];  //将数组中的每一个元素的下标改成随机数
    arr[i] = temp;     //将元素调换位置
}
console.log(arr)
```
