##### 四则运算
```js
function count(x,y,sign){
    if(sign=="+"){
        console.log(`${x}+${y}=${x+y}`)
    }
    if(sign=="-"){
        console.log(`${x}-${y}=${x-y}`)
    }
    if(sign=="*"){
        console.log(`${x}*${y}=${x*y}`)
    }
    if(sign=="/"){
        console.log(`${x}/${y}=${x/y}`)
    }
}
count(2,5,"+")
count(2,5,"-")
count(2,5,"*")
count(2,5,"/")
```

##### 数组排序 升序（冒泡）
```js
function sort(arr){
    for(let i=0;i<arr.length-1;i++){
        for(let j=0;j<arr.length-1-i;j++){
            if(arr[j]>arr[j+1]){
                let temp=arr[j]
                arr[j]=arr[j+1]
                arr[j+1]=temp
            }
        }
    }
    console.log(arr)
}
sort([48,7,56,5,32,7,9,88,0])
```

```js
function splice(arr,index,length,...num){
    let arr1=[]
    for(let i=0;i<arr.length;i++){
        if(i<index){
            arr1[arr1.length]=arr[i]
        }else if(i==index){
            for(let j=0;j<length;j++){
                i++
            }
            for(let k=0;k<num.length;k++){
                arr1[arr1.length]=num[k]
            }
            if(arr[i]!=null){
                arr1[arr1.length]=arr[i]
            }
        }else{
            arr1[arr1.length]=arr[i]
        }
    }
    console.log(arr1)

}
```

##### 5.函数index(arr,item)返回item在arr中的下标
```js
function index(arr,item){
    for(i=0;i<arr.length;i++){
        if(arr[i]==item){
            return i
        }
    }
    return (`该数组中没有${item}`)
}
```

##### 7.concat(arr1,arr2)
```js
let arr1=[4,5,8,6,9,3]
let arr2=[12,58,4,7,6]
function concat(arr1,arr2){
  let newArr=[]
  newIndex=0
  for(let i=0;i<arr1.length;i++){
    newArr[newIndex]=arr1[i]
    newIndex++
  }
  for(let j=0;j<arr2.length;j++){
    newArr[newIndex]=arr2[j]
    newIndex++
  }
  return newArr
}
console.log(concat(arr1,arr2))
```

##### 递归函数  求阶乘
```js
function jiecheng(num){
    if(num===1){
        return 1;
    }
    else{
        return num*jiecheng(num-1);
    }
}
let num=prompt("请输入一个数","5");
console.log(jiecheng(num));
```
